## **链表反转**
{% highlight python %}
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
//非递归方法：需要三个指针完成，要注意链表断裂的考虑
class Solution {
public:
    ListNode* ReverseList(ListNode* pHead) {
        if(pHead==nullptr || pHead->next==nullptr)
            return pHead;
        ListNode* pNext=pHead->next;
        ListNode* pReverse=pHead;
        ListNode* pPre=pHead;
        while(pNext->next != nullptr)
        {
            pReverse=pNext;
            pNext=pNext->next;
            pReverse->next=pPre;
            pPre=pReverse;
        }
        pNext->next=pPre;
        pHead->next=nullptr;
        return pNext;
    }
};
{% endhighlight %}
